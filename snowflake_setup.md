# Introduction and Environment Setup

## Snowflake user creation
Copy these SQL statements into a Snowflake Worksheet, select all and execute them (i.e. pressing the play button).

```sql
-- Use an admin role
USE ROLE ACCOUNTADMIN;

-- Create the `transform` role
CREATE ROLE IF NOT EXISTS transform;
GRANT ROLE TRANSFORM TO ROLE ACCOUNTADMIN;

-- Create the default warehouse if necessary
CREATE WAREHOUSE IF NOT EXISTS COMPUTE_WH;
GRANT OPERATE ON WAREHOUSE COMPUTE_WH TO ROLE TRANSFORM;

-- Create our database and schemas
CREATE DATABASE IF NOT EXISTS PAYROLL_WAREHOUSE;
CREATE SCHEMA IF NOT EXISTS PAYROLL_WAREHOUSE.LANDING;
CREATE SCHEMA IF NOT EXISTS PAYROLL_WAREHOUSE.MARTS;


-- Create the `dbt` user and assign to role
CREATE USER IF NOT EXISTS dbt_payroll
  PASSWORD='dbtPayroll123'
  LOGIN_NAME='dbt_payroll'
  MUST_CHANGE_PASSWORD=FALSE
  DEFAULT_WAREHOUSE='COMPUTE_WH'
  DEFAULT_ROLE='transform'
  DEFAULT_NAMESPACE='PAYROLL_WAREHOUSE.LANDING'
  COMMENT='DBT user used for data transformation';
GRANT ROLE transform to USER dbt_payroll;

-- Set up permissions to role `transform`
GRANT ALL ON WAREHOUSE COMPUTE_WH TO ROLE transform; 
GRANT ALL ON DATABASE PAYROLL_WAREHOUSE to ROLE transform;
GRANT ALL ON ALL SCHEMAS IN DATABASE PAYROLL_WAREHOUSE to ROLE transform;
GRANT ALL ON FUTURE SCHEMAS IN DATABASE PAYROLL_WAREHOUSE to ROLE transform;
GRANT ALL ON ALL TABLES IN SCHEMA PAYROLL_WAREHOUSE.LANDING to ROLE transform;
GRANT ALL ON FUTURE TABLES IN SCHEMA PAYROLL_WAREHOUSE.LANDING to ROLE transform;


-- Setup permissions for pbi user
CREATE ROLE IF NOT EXISTS REPORTER;
CREATE USER IF NOT EXISTS PBI
    PASSWORD='pbiPassword123'
    LOGIN_NAME='pbi'
    MUST_CHANGE_PASSWORD=FALSE
    DEFAULT_WAREHOUSE='COMPUTE_WH'
    DEFAULT_ROLE='REPORTER'
    DEFAULT_NAMESPACE='PAYROLL_WAREHOUSE.MARTS'
    COMMENT='PowerBI user for creating reports';
GRANT ROLE REPORTER TO USER PBI;
GRANT ROLE REPORTER TO ROLE ACCOUNTADMIN;
GRANT ALL ON WAREHOUSE COMPUTE_WH TO ROLE REPORTER;
GRANT USAGE ON DATABASE PAYROLL_WAREHOUSE TO ROLE REPORTER;
GRANT USAGE ON SCHEMA PAYROLL_WAREHOUSE.MARTS TO ROLE REPORTER;
GRANT SELECT ON ALL TABLES IN SCHEMA PAYROLL_WAREHOUSE.MARTS TO ROLE REPORTER;
GRANT SELECT ON FUTURE TABLES IN SCHEMA PAYROLL_WAREHOUSE.MARTS TO ROLE REPORTER;

--setup schema and s3 if load from S3/processed to snowflake LANDING DB INSTEAD OF USING SPARK(if computer --memory is only 8GB
CREATE STORAGE INTEGRATION s3_payroll
    TYPE = EXTERNAL_STAGE
    STORAGE_PROVIDER = S3
    ENABLED = TRUE
    STORAGE_AWS_ROLE_ARN = "arn:aws:iam::869935110009:role/snowflake_payroll" --save to env
    STORAGE_ALLOWED_LOCATIONS = ('s3://payroll-project/'); --save to env

DESC INTEGRATION s3_payroll
